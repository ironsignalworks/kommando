function setKeys(nx, ny){ // When the render is rotated +90° (portrait), rotate input −90° back to world space. // screen -> world: [wx, wy] = [-ny, nx] let wx = nx, wy = ny; if (isPortraitMobile()){ wx = ny; wy = -nx; } const want = { ArrowLeft: wx < -DEAD, ArrowRight: wx > DEAD, ArrowUp: wy < -DEAD, ArrowDown: wy > DEAD }; for(const k in want){ if(want[k] && !pressed[k]){ fire(k,'keydown'); pressed[k]=true; } if(!want[k] && pressed[k]){ fire(k,'keyup'); pressed[k]=false; } } }
